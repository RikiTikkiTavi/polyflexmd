Define rouse model¶

In [61]:
def rouse_R_sq_avg(t, N: int, l: float):
    return np.repeat(N * l ** 2, t.shape)

def rouse_R_autocorr(t, p_max: int, N: int, l: float, tau_1: float):
    result = 0
    for p in range(1, p_max):
        if p % 2 == 0:
            continue
        result += 1 / p ** 2 * np.exp(-t * p ** 2 / tau_1)
    return 8 * N * l ** 2 / np.pi ** 2 * result

def rouse_g_4(t, tau_1: float, p_max: int, N: int, l: float):
    s = 0
    for p in range(1, p_max + 1):
        if p % 2 == 0:
            continue
        s += 1 / p ** 2 * np.exp(-t * p ** 2 / tau_1)
    return 2 * N * l ** 2 * (1 - 8 / np.pi ** 2 * s)
